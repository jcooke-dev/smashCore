110    def next_level(self) -> None:
111        """
112        Builds the next level, resets the ball position and initial speed
113        Slight increase in initial ball speed to add difficulty
114        (NOTE this doesn't actually increment the level num)
115        
116        :return:
117        """
118        self.gw.remove_obstacles()
119        self.gw.remove_bricks()
120
121        for wo in self.gw.world_objects:
122            if isinstance(wo, Ball):
123                wo.reset_position()
124                wo.speed_v = BALL_SPEED_VECTOR + (self.ps.level * BALL_SPEED_LEVEL_INCREMENT)
125                self.gs.ball_speed_increased_ratio = wo.speed_v / BALL_SPEED_VECTOR
126                wo.v_vel = wo.v_vel_unit * wo.speed_v
127                wo.speed = BALL_SPEED_SIMPLE + (self.ps.level * BALL_SPEED_LEVEL_INCREMENT)
128        # builds the next level (NOTE this doesn't actually increment the level num)
129        next_level = Levels.get_level_name_from_num(self.ps.theme, self.ps.level)
130        Levels.build_level(self.gw.world_objects, next_level)
131        self.gs.level_cleared = False
132
133        self.fps = INITIAL_FPS_SIMPLE
134        self.gs.cur_state = GameState.GameStateName.READY_TO_LAUNCH

136    def set_graphics_mode(self) -> None:
137        """
138        Handles the pygame.display mode setting so that we can swap between windowed and fullscreen.
139
140        :return:
141        """
142
143        # best to quit and then start fresh with set_mode() (helps avoid the exception with failed renderer
144        # from using SCALED flag)
145        pygame.display.quit()
146
147        try:
148            # create the FULLSCREEN with SCALED flag to fix the slightly off-center display
149            self.screen = pygame.display.set_mode((WIDTH, HEIGHT),
150                                                  flags=0 if not self.gset.is_fullscreen else (pygame.FULLSCREEN | pygame.SCALED) )
151        except Exception as ex:
152            # if problem with renderer, just exclude the SCALED flag as a fallback
153            pygame.display.quit()
154            self.screen = pygame.display.set_mode((WIDTH, HEIGHT),
155                                                  flags=0 if not self.gset.is_fullscreen else pygame.FULLSCREEN)
156
157        self.ui.screen = self.screen
158        pygame.display.set_caption(GAME_NAME)

200    def play_music(self):
201        """
202        Plays the music file for each game state
203        
204        :return:
205        """
206        if not self.gset.bgm_sounds:
207            pygame.mixer.music.stop()
208            self.current_music_path = None
209            return
210
211        target_music_path: str = ""
212        loop: int = -1  # Default to loop infinitely
213
214        if self.gs.cur_state in assets.MUSIC_PATHS:
215            target_music_path = assets.MUSIC_PATHS[self.gs.cur_state]
216            if ((self.gs.cur_state == GameState.GameStateName.SPLASH) or
217                    (self.gs.cur_state == GameState.GameStateName.GET_HIGH_SCORE) or
218                    (self.gs.cur_state == GameState.GameStateName.GAME_OVER)):
219                loop = 0  # Play only once
220
221        if target_music_path and self.current_music_path != target_music_path:
222            pygame.mixer.music.stop()
223            pygame.mixer.music.load(target_music_path)
224            pygame.mixer.music.set_volume(self.gset.music_volume)
225            pygame.mixer.music.play(loop)
226            self.current_music_path = target_music_path
227        elif not target_music_path and self.current_music_path is not None:
228            pygame.mixer.music.stop()
229            self.current_music_path = None

527    def handle_collisions_between_worldobjects(self, current_wo, other_wo):
528        """
529        Handle collisions between world objects
530        :param current_wo:
531        :param other_wo:
532        :return:
533        """
534        if current_wo.rect.colliderect(other_wo.rect):
535            # a collision was detected - should we react to it?  this matters because two
536            # objects can overlap/collide across multiple looping collision checks - if
537            # we don't deactivate the collision detection, the object can bounce back and
538            # forth, getting trapped
539            if other_wo.allow_collision():
540                # bounce object properly -
541                # determining in which direction
542                # to bounce, based on approach
543                current_wo.detect_collision(other_wo, self.gs, self.gset)
544                other_wo.add_collision(self.gset)
545                if other_wo.should_score():
546                    self.ps.score += other_wo.value
547                if other_wo.should_remove():
548                    self.ps.score += other_wo.bonus
549
550                    # trigger the special effect - the Brick adds the appropriate Animation object to the world
551                    other_wo.trigger_destruction_effect(self.gw.world_objects, self.gset, self.ps)
552
553                    # if this Brick is strong enough for the shake, get that started
554                    if other_wo.strength_initial >= SHAKE_STRENGTH_THRESHOLD:
555                        utils.start_shake(self.gs, other_wo.strength_initial * SHAKE_OFFSET_BASE)
556
557                    # now remove the actual Brick object
558                    self.gw.world_objects.remove(other_wo)
559
560                    current_wo.speed += .20
561                    # BALL_SPEED_STEP: adding to the ball speed, but diff logic for the
562                    # VECTOR models
563                    if isinstance(current_wo, Ball):
564                        current_wo.speed_v += self.gs.ball_speed_step
565                        self.gs.ball_speed_increased_ratio = current_wo.speed_v / BALL_SPEED_VECTOR
566                        current_wo.v_vel = current_wo.v_vel_unit * current_wo.speed_v
567
568        else:
569            # this is the other side of the allow_collision logic above, since
570            # not colliding now, it resets the latch or 'primed for collision' flag
571            other_wo.prime_for_collision()

575        # event handling
576        ##############################################################

603                        self.gs.show_dev_overlay = not self.gs.show_dev_overlay

605                # detect the CTRL+a key combo to toggle the AUTO-PLAY mode
606                # on and off
607                if event.key == pygame.K_a:

620                            self.gs.paddle_impulse_vel_length += PADDLE_IMPULSE_INCREMENT

623                # increase/decrease the WORLD_GRAVITY_ACC

626                        if event.mod & pygame.KMOD_SHIFT:
627                            self.gs.gravity_acc_length -= WORLD_GRAVITY_ACC_INCREMENT
628                            if self.gs.gravity_acc_length < 0.0:
629                                self.gs.gravity_acc_length = 0.0

631                        else:
632                            self.gs.gravity_acc_length += WORLD_GRAVITY_ACC_INCREMENT

635                # detect the CTRL+s and CTRL+SHIFT+s key combos to
636                # increase/decrease the BALL_SPEED_STEP
637                if event.key == pygame.K_s:
638                    if event.mod & pygame.KMOD_CTRL:
639                        if event.mod & pygame.KMOD_SHIFT:
640                            self.gs.ball_speed_step -= BALL_SPEED_STEP_INCREMENT
641                        else:
642                            self.gs.ball_speed_step += BALL_SPEED_STEP_INCREMENT
643
644                # detect the CTRL+m key combo to cycle through the various motion models
645                if event.key == pygame.K_m:
646                    if event.mod & pygame.KMOD_CTRL:
647                        match self.gs.motion_model:
648                            case MotionModels.SIMPLE_1:
649                                self.gs.motion_model = MotionModels.VECTOR_1
650                            case MotionModels.VECTOR_1:
651                                self.gs.motion_model = MotionModels.SIMPLE_1

653                # detect the CTRL+'=' and CTRL+'-' key combos to adjust music volume
654                if event.key == pygame.K_EQUALS:
655                    if event.mod & pygame.KMOD_CTRL:
656                        if event.mod & pygame.KMOD_SHIFT:
657
658                            old_sfx_vol = self.gset.sfx_volume
659
660                            self.gset.sfx_sounds = True
661                            self.gset.sfx_volume += SFX_VOLUME_STEP
662                            self.gset.sfx_volume = min(self.gset.sfx_volume, 1.0)
663
664                            # check for a changed SFX volume, if so, play a sample sound
665                            if (self.gs.cur_state == GameState.GameStateName.SETTINGS) and (abs(self.gset.sfx_volume - old_sfx_vol) > CLOSE_TO_ZERO):
666                                snd: pygame.mixer.Sound = pygame.mixer.Sound(assets.PADDLE_SFX)
667                                snd.set_volume(self.gset.sfx_volume)

670                        else:
671                            self.gset.bgm_sounds = True
672                            self.gset.music_volume += MUSIC_VOLUME_STEP
673                            self.gset.music_volume = min(self.gset.music_volume, 1.0)
674                            pygame.mixer.music.set_volume(self.gset.music_volume)

676                # detect the CTRL+'+' and CTRL+'-' key combos to adjust music volume
677                if event.key == pygame.K_MINUS:
678                    if event.mod & pygame.KMOD_CTRL:
679                        if event.mod & pygame.KMOD_SHIFT:
680
681                            old_sfx_vol = self.gset.sfx_volume
682
683                            self.gset.sfx_volume -= SFX_VOLUME_STEP
684                            self.gset.sfx_volume = max(self.gset.sfx_volume, 0.0)
685                            if abs(self.gset.sfx_volume) < CLOSE_TO_ZERO:
686                                self.gset.sfx_sounds = False
687
688                            # check for a changed SFX volume, if so, play a sample sound
689                            if (self.gs.cur_state == GameState.GameStateName.SETTINGS) and (abs(self.gset.sfx_volume - old_sfx_vol) > CLOSE_TO_ZERO):
690                                snd: pygame.mixer.Sound = pygame.mixer.Sound(assets.PADDLE_SFX)
691                                snd.set_volume(self.gset.sfx_volume)
692                                pygame.mixer.find_channel(True).play(snd)
693
694                        else:
695                            self.gset.music_volume -= MUSIC_VOLUME_STEP
696                            self.gset.music_volume = max(self.gset.music_volume, 0.0)
697                            if abs(self.gset.music_volume) < CLOSE_TO_ZERO:
698                                self.gset.bgm_sounds = False
699                            pygame.mixer.music.set_volume(self.gset.music_volume)
700
701                # detect the CTRL+l to force-load next level in sequence
702                if event.key == pygame.K_l:
703                    if event.mod & pygame.KMOD_CTRL:
704                        self.ps.level += 1
705                        self.next_level()
706
707                # handle initials textbox input
708                if self.gs.cur_state == GameState.GameStateName.GET_HIGH_SCORE:
709                    if (event.key == pygame.K_RETURN) or (event.key == pygame.K_KP_ENTER):
710                        self.lb.add_score(self.ps, self.ui)
711                        self.gs.cur_state = GameState.GameStateName.GAME_OVER
712                    elif event.key == pygame.K_BACKSPACE:
713                        self.ui.tb_initials_text = self.ui.tb_initials_text[:-1]
714                    elif len(self.ui.tb_initials_text) < SCORE_INITIALS_MAX:
715                        self.ui.tb_initials_text += event.unicode
716
717            # the actual button press checks from the returned rects above
718            if (event.type == pygame.MOUSEBUTTONDOWN and
719                    ((self.gs.cur_state == GameState.GameStateName.PAUSED) or
720                     (self.gs.cur_state == GameState.GameStateName.GAME_OVER))):
721                if self.restart_game_button.collidepoint(event.pos):
722                    self.reset_game()
723                if self.quit_game_button.collidepoint(event.pos):
724                    self.clean_shutdown()
725                if self.main_menu_button.collidepoint(event.pos):
726                    self.gs.cur_state = GameState.GameStateName.MENU_SCREEN
727
728            if (event.type == pygame.MOUSEBUTTONDOWN and
729                    (self.gs.cur_state == GameState.GameStateName.GET_HIGH_SCORE)):
730                if self.high_score_enter_btn.collidepoint(event.pos):
731                    self.lb.add_score(self.ps, self.ui)
732                    self.gs.cur_state = GameState.GameStateName.GAME_OVER

734    def run_loop(self) -> None:
735        """
736        Runs the main game loop
737
738        :return:
739        """
740
741        while self.gs.running:
742            # fill the screen with black as a good default
743            self.screen.fill(BLACK)
744            self.play_music()
745
746            # get all events from queue for handling
747            events = pygame.event.get()
748
749            self.handle_gamestate(events)
750
751            self.handle_events(events)
752
753            # get the continuously pressed keys, rather than single key press events
754            pressed_keys = pygame.key.get_pressed()
755            if not self.gset.paddle_under_mouse_control or self.gset.paddle_under_auto_control:
756                if pressed_keys[pygame.K_LEFT]:
757                    self.gs.paddle_under_key_control_left = True
758                elif pressed_keys[pygame.K_RIGHT]:
759                    self.gs.paddle_under_key_control_right = True
760
761            # draw the developer overlay, if requested
762            if self.gs.show_dev_overlay:
763                self.ui.draw_dev_overlay(self.gs)
764
765            ##############################################################
766            # update screen
767            ##############################################################
768            pygame.display.flip()
769
770            # choose from the available motion models; note that SIMPLE models
771            # use clock.tick(fps) to force the motion update logic to the
772            # frame rate - VECTOR models decouple the frame rate from the
773            # dT motion logic
774            if self.gs.motion_model == MotionModels.SIMPLE_1:
775                self.gs.tick_time = self.clock.tick(self.fps)
776            elif self.gs.motion_model == MotionModels.VECTOR_1:
777                # removing the fps arg (rather, setting it to 0) allows pygame
778                # to run this loop at full speed
779                # self.gs.tick_time = self.clock.tick(MAX_FPS_VECTOR)
780                self.gs.tick_time = self.clock.tick_busy_loop(MAX_FPS_VECTOR)
781
782            # don't bother calculating these running dev averages unless wanted
783            if self.gs.show_dev_overlay:
784                self.gs.fps_avg, self.gs.loop_time_avg = utils.calculate_timing_averages(self.clock.get_fps(),
785                                                                                         self.clock.get_time())
786
787        ##############################################################
788        # close down cleanly
789        ##############################################################
790        pygame.quit()

528    def handle_collisions_between_worldobjects(self, current_wo, other_wo):
529        """
530        Handle collisions between world objects
531        :param current_wo:
532        :param other_wo:
533        :return:
534        """
535        if current_wo.rect.colliderect(other_wo.rect):
536            # a collision was detected - should we react to it?  this matters because two
537            # objects can overlap/collide across multiple looping collision checks - if
538            # we don't deactivate the collision detection, the object can bounce back and
539            # forth, getting trapped
540            if other_wo.allow_collision():
541                # bounce object properly -
542                # determining in which direction
543                # to bounce, based on approach
544                current_wo.detect_collision(other_wo, self.gs, self.gset)
545                other_wo.add_collision(self.gset)
546                if other_wo.should_score():
547                    self.ps.score += other_wo.value
548                if other_wo.should_remove():
549                    self.ps.score += other_wo.bonus
550
551                    # trigger the special effect - the Brick adds the appropriate Animation object to the world
552                    other_wo.trigger_destruction_effect(self.gw.world_objects, self.gset, self.ps)
553
554                    # if this Brick is strong enough for the shake, get that started
555                    if other_wo.strength_initial >= SHAKE_STRENGTH_THRESHOLD:
556                        utils.start_shake(self.gs, other_wo.strength_initial * SHAKE_OFFSET_BASE)
557
558                    # now remove the actual Brick object
559                    self.gw.world_objects.remove(other_wo)
560
561                    current_wo.speed += .20
562                    # BALL_SPEED_STEP: adding to the ball speed, but diff logic for the
563                    # VECTOR models
564                    if isinstance(current_wo, Ball):
565                        current_wo.speed_v += self.gs.ball_speed_step
566                        self.gs.ball_speed_increased_ratio = current_wo.speed_v / BALL_SPEED_VECTOR
567                        current_wo.v_vel = current_wo.v_vel_unit * current_wo.speed_v
568
569        else:
570            # this is the other side of the allow_collision logic above, since
571            # not colliding now, it resets the latch or 'primed for collision' flag
572            other_wo.prime_for_collision()
