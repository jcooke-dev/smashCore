 75    def draw_button(self, btn_surface: pygame.Surface, x: int, y: int, width: int, height: int, color: pygame.color,
 76                    hover_color: pygame.color, action: Callable = None, corner_radius: int = 10) -> pygame.Rect:
 77        """
 78        Draw a button and return its Rect.
 79
 80        :param corner_radius:
 81        :param btn_surface: button text
 82        :param x: Rect x
 83        :param y: Rect y
 84        :param width: Rect width
 85        :param height: Rect height
 86        :param color: button color
 87        :param hover_color: button hover color
 88        :param action: Function to call on click
 89        :return: pygame.Rect of the button
 90        """
 91        mouse = pygame.mouse.get_pos()
 92        click = pygame.mouse.get_pressed()
 93        rect = pygame.Rect(x, y, width, height)
 94
 95        if rect.collidepoint(mouse):
 96            pygame.draw.rect(self.surface, hover_color, rect, border_radius=corner_radius)
 97            if click[0] == 1 and action is not None:
 98                action()
 99        else:
100            pygame.draw.rect(self.surface, color, rect, border_radius=corner_radius)
101
102        text_rect = btn_surface.get_rect(center=rect.center)
103        self.surface.blit(btn_surface, text_rect)
104        return rect

417    def update_background_elements(self):
418        """Collisions for menu screen"""
419        for ball in self.background_balls:
420            ball['rect'].x += ball['speed_x']
421            ball['rect'].y += ball['speed_y']
422            if ball['rect'].left < 0 or ball['rect'].right > constants.WIDTH:
423                ball['speed_x'] *= -1
424                ball['speed_y'] += random.uniform(-0.5, 0.5)
425            if ball['rect'].top < 0 or ball['rect'].bottom > constants.HEIGHT:
426                ball['speed_y'] *= -1
427                ball['speed_x'] += random.uniform(-0.5, 0.5)
428
429            for brick in self.background_bricks:
430                if ball['rect'].colliderect(brick['rect']):
431                    overlap_x = min(ball['rect'].right, brick['rect'].right) - \
432                                max(ball['rect'].left, brick['rect'].left)
433                    overlap_y = min(ball['rect'].bottom, brick['rect'].bottom) - \
434                                max(ball['rect'].top, brick['rect'].top)
435
436                    if overlap_x > 0 and overlap_y > 0:
437                        if abs(overlap_x) < abs(overlap_y):
438                            if ball['speed_x'] > 0:
439                                ball['rect'].right = brick['rect'].left
440                            else:
441                                ball['rect'].left = brick['rect'].right
442                            ball['speed_x'] *= -1
443                        else:
444                            if ball['speed_y'] > 0:
445                                ball['rect'].bottom = brick['rect'].top
446                            else:
447                                ball['rect'].top = brick['rect'].bottom
448                            ball['speed_y'] *= -1

 76    def draw_button(self, btn_surface: pygame.Surface, x: int, y: int, width: int, height: int, color: pygame.color,
 77                    hover_color: pygame.color, action: Callable = None, corner_radius: int = 10) -> pygame.Rect:
 78        """
 79        Draw a button and return its Rect.
 80
 81        :param corner_radius:
 82        :param btn_surface: button text
 83        :param x: Rect x
 84        :param y: Rect y
 85        :param width: Rect width
 86        :param height: Rect height
 87        :param color: button color
 88        :param hover_color: button hover color
 89        :param action: Function to call on click
 90        :return: pygame.Rect of the button
 91        """
 92        mouse = pygame.mouse.get_pos()
 93        click = pygame.mouse.get_pressed()
 94        rect = pygame.Rect(x, y, width, height)
 95
 96        if rect.collidepoint(mouse):
 97            pygame.draw.rect(self.surface, hover_color, rect, border_radius=corner_radius)
 98            if click[0] == 1 and action is not None:
 99                action()
100        else:
101            pygame.draw.rect(self.surface, color, rect, border_radius=corner_radius)
102
103        text_rect = btn_surface.get_rect(center=rect.center)
104        self.surface.blit(btn_surface, text_rect)
105        return rect

418    def update_background_elements(self):
419        """Collisions for menu screen"""
420        for ball in self.background_balls:
421            ball['rect'].x += ball['speed_x']
422            ball['rect'].y += ball['speed_y']
423            if ball['rect'].left < 0 or ball['rect'].right > constants.WIDTH:
424                ball['speed_x'] *= -1
425                ball['speed_y'] += random.uniform(-0.5, 0.5)
426            if ball['rect'].top < 0 or ball['rect'].bottom > constants.HEIGHT:
427                ball['speed_y'] *= -1
428                ball['speed_x'] += random.uniform(-0.5, 0.5)
429
430            for brick in self.background_bricks:
431                if ball['rect'].colliderect(brick['rect']):
432                    overlap_x = min(ball['rect'].right, brick['rect'].right) - \
433                                max(ball['rect'].left, brick['rect'].left)
434                    overlap_y = min(ball['rect'].bottom, brick['rect'].bottom) - \
435                                max(ball['rect'].top, brick['rect'].top)
436
437                    if overlap_x > 0 and overlap_y > 0:
438                        if abs(overlap_x) < abs(overlap_y):
439                            if ball['speed_x'] > 0:
440                                ball['rect'].right = brick['rect'].left
441                            else:
442                                ball['rect'].left = brick['rect'].right
443                            ball['speed_x'] *= -1
444                        else:
445                            if ball['speed_y'] > 0:
446                                ball['rect'].bottom = brick['rect'].top
447                            else:
448                                ball['rect'].top = brick['rect'].bottom
449                            ball['speed_y'] *= -1
